<head>
    <!--css优先级 行内>外部>内部  id选择器>类选择器>标签选择器 -->
    <title>css</title>
    <!-- css内部样式-->
    <style type="text/css">
        /* 标签选择器*/
        h1{
            /* 颜色支持英文和16进制*/                
            font-size:30px;
        }
        p{
            color:blue;
        }
        /*类选择器 可以用于多个*/
        .mystyle
        {
            background-color: green;
            color:yellow; 
        }
        /* id选择器 一次只能选择一个*/
        #mystyle2
        {
            background-color: yellow;
        }
        /* 后代选择器*/
        ul strong
        {
            color: red;
        }
        /*交集选择器不能继承，连续书写 */
        li.mystyle3
        {
            color: green;
        }
        /*并集选择器 */
        ul,h1
        {
            color: blue;
        }
        /*层次选择器*/
        ul>li>strong
        {
            color: red;
        }
        /*相邻同辈选择器（后面）只对h1相邻的h2有效*/
        h4+h5
        {
            background-color: purple;
        }
        /*通用同辈选择器（只对后面的有效）h1后的所有h2*/
        h4~h5
        {
            background-color: red;
        }
        /*伪类选择器 在火狐用不了不知道是不是写错了
        li : first-child
        {
            background-size: 3px;
        }
        li : last-child
        {
            background-color: blue;
        }
        li : nth-child(2)
        {
            background-color: red;
        } 
        li: first-of-type
        {
            background-color: purple;
        }
        li: nth-of-type(2)
        {

        } */
        /*选择有input和name*/
        /*以u开头*/
        input[name^=u]
        {
            background-color: yellow;
        }
        /*以na结尾*/ 
        input[name$=na]
        {

        }
        /*有name属性*/
        input[name]
        {

        }
        /*name等于username*/
        input[name=username]
        {

        }
        /*包含rn*/
        input[name*=rn]
        {

        }
