<!-- 这个可以省略，是一个声明-->
<!DOCTYPE html>
<!-- 这是最外层html必须包裹在html标签里面，所有标签不区分大小写-->
<html>
    <!-- 设置参数，不能显示只有title可以-->
    <head>
        <!--标题-->
        <title>first html</title>
        <!-- 设置编码-->
        <meta charset="utf-8">
        <!-- 关键词 是否能被搜索引擎搜到-->
        <meta name="keywords" content="学习html">
        <!-- 搜到本网页时展示的简介内容-->
        <meta name="description" content="这是一个学习html的网页">
        <!--css-->
        <style type="text/css">
            h1{
                color:red;
            }
        </style>
    </head>
    <!-- 能显示-->
    <body>
           <!--标题标签 越来越小-->
           <h1>一级标题</h1>
           <h1>一级标题</h1>
           <h2>二级标题</h2>
           <h3>三级标题</h2>
           <h4>四级标题</h2>
           <h5>五级标题</h5>
           <!-- 回车和空格没有用 <br/> 强行回车-->
           fsdfsadf<br/>  fsdfsd
           <!--段落标签末尾自带回车-->
           <p>fsdfsd fsdfsd</p>
           fsdfsdf
           <!-- <hr/>水平线-->
           <hr/>水平线
           fdsfsd
           <!--样式标签，可以嵌套-->
           <strong>加粗</strong>
           <b>加粗</b>
           <em>hello倾斜</em>
           <strong><em>加粗倾斜</em></strong>
           <!--特殊符号-->
           <!--&lt代表< &gt代表> -->
           你好&lt&gt<br>
           <!--&nbsp;代表空格 -->
           你好&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;啊
           <!--&quot;代表" -->
           &quot;你好&quot;
           <!-- &copy;代表版权-->
           <br>&copy;2020
           <!--图片一般用相对路径 -->
           <img  src="a/1.jpg" width="100" height="100" title="天海祐希（鼠标悬浮文字）"
           alt="图片不存在时的替换路径">
           <!-- 超链接-->
           <a href="https://www.baidu.com">点我进百度</a>
           <!-- 锚链接跳到指定位置#后写name位置-->
           <a href="html-test2.html#2">进入2</a>
           <!--功能性链接 -->
           <a href="mailto:1829731045@qq.com">发邮件</a>
           <!-- 列表无序有序-->
           <ul type="square">
               <li>apple</li>
               <li>banana</li>
               <li>orange</li>
           </ul>
