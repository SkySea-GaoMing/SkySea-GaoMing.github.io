<style type="text/css">
    .m
    {
        /*空间向左压缩使下面的元素上移*/
        float:left;
        /*清除浮动还原换行特性*/
        clear:left;
    }
    div p
    {
        width:100px;
        height:100px;
        border:solid red 3px;
        /*隐藏溢出的内容*/
    /*    overflow:hidden; */
        /*滚动条*/
        overflow:scroll;
        /*定位相对自身*/
        position:relative;
        left:50px;
        top:100px;
        /*绝对定位参照浏览器,或者已经定位的父元素，会释放空间*/
    /*    position:absolute;*/
        /*一直悬浮在固定位置参照浏览器 */
        position:fixed;
        /*透明度*/
        opacity:0.5;

    }
    img
    {
        /*图片居中*/
        vertical-align:middle;
    }
    a:hover
    {
        /*变型平移*/
        transform:translate();
    }
</style>
